        void String(char str[]) {
            int i = 0;
            while (str[i]) {
                if (str[i] >= 'a' && str[i] <= 'z')
                    str[i] = 'A' + (str[i] - 'a');
                else if (str[i] >= 'A' && str[i] <= 'Z')

                i++;
            }
